// 1. Initialize Map const map = L.map('map').setView([49.8951, -97.1384], 12); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { attribution: '© OpenStreetMap contributors' }).addTo(map); // Global State let parsedData = []; // The raw data from the file let markers = []; // The actual Leaflet marker objects // --- 2. The Fuzzy Header Matcher --- function standardizeHeaders(row) { const idAliases = ['id', 'name', 'title', 'location', 'stop', 'customer']; const latAliases = ['lat', 'latitude', 'y', 'lat.', 'latitude_decimal']; const lngAliases = ['lng', 'lon', 'longitude', 'x', 'long', 'longitude_decimal']; let cleanRow = { id: "Unknown", lat: null, lng: null }; for (const [key, value] of Object.entries(row)) { const lowerKey = key.trim().toLowerCase(); if (idAliases.includes(lowerKey)) cleanRow.id = value; if (latAliases.includes(lowerKey)) cleanRow.lat = parseFloat(value); if (lngAliases.includes(lowerKey)) cleanRow.lng = parseFloat(value); } return cleanRow; } // --- 3. File Input Listener --- document.getElementById('fileInput').addEventListener('change', function(e) { const file = e.target.files[0]; Papa.parse(file, { header: true, skipEmptyLines: true, complete: function(results) { parsedData = results.data.map(standardizeHeaders).filter(r => r.lat && r.lng); renderPreview(); } }); }); function renderPreview() { const list = document.getElementById('previewList'); list.innerHTML = parsedData.map((p, i) => `
  • ${i === 0 ? '🟢 ' : ''}${p.id} ${p.lat.toFixed(4)}, ${p.lng.toFixed(4)}
    • `).join(''); document.getElementById('btnLoadMap').classList.remove('hidden'); } // --- 4. NEW: Load to Map Logic --- document.getElementById('btnLoadMap').addEventListener('click', function() { // Clear existing markers markers.forEach(m => map.removeLayer(m)); markers = []; const bounds = []; parsedData.forEach((point, index) => { const marker = L.marker([point.lat, point.lng], { draggable: true, title: point.id }).addTo(map); // Add a popup so you can rename or see the ID marker.bindPopup(`${point.id}
    Stop #${index}`); // Update data if you drag the pin! marker.on('dragend', function(event) { const newPos = event.target.getLatLng(); parsedData[index].lat = parseFloat(newPos.lat.toFixed(6)); parsedData[index].lng = parseFloat(newPos.lng.toFixed(6)); renderPreview(); // Sync the sidebar }); markers.push(marker); bounds.push([point.lat, point.lng]); }); // Zoom the map to fit all your pins if (bounds.length > 0) { map.fitBounds(bounds, { padding: [50, 50] }); } // Enable the Optimize button const optBtn = document.getElementById('btnOptimize'); optBtn.disabled = false; optBtn.classList.remove('opacity-50', 'cursor-not-allowed'); }); // --- 5. NEW: The API Call (Trigger Optimization) --- document.getElementById('btnOptimize').addEventListener('click', async function() { const btn = this; btn.innerText = "Optimizing..."; try { const response = await fetch('/api/optimize', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ locations: parsedData }) }); const result = await response.json(); if (result.status === 'success') { alert(`Optimized! Total Distance: ${result.metrics.distance_km} km`); // We'll add the "Draw Route Line" logic next! console.log("Ordered Route:", result.optimized_route); } } catch (err) { alert("Optimization failed. Check console."); console.error(err); } finally { btn.innerText = "Optimize Route"; } });
